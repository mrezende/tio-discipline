\subsection{Definição}

\begin{frame}
\frametitle{\textbf{Inovação revolucionária}\cite{mascitelli:00}}
	\begin{itemize}
	\item \textit{"What makes an \textbf{innovation disruptive}, according to Christensen, is that it captures new markets by combining performance attributes	in an original and unexpected way"}
	\item \textit{"innovations that are unique, original, and unexpected are far more valuable from a competitive standpoint than innovations that are predictable, incremental, or mundane"}
	\item \textit{"...the term \textbf{breaktrough innovation} hereafter to represent any creative and original action by individuals or project teams that enable firms to capture at least temporary monopoly profits or that results in a significant increase in market share"}
	\end{itemize}
\end{frame}
